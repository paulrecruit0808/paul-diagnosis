  const c=COURSES[currentCourse],t=TYPES[currentType];
  const goal=c.typeGoals?.[currentType]||t.goal;
  const msg=TYPE_COURSE_MSG?.[currentType]?.[currentCourse]||'';
  const joinInfo=parseJoinMonth(profile.joinMonth);
  document.getElementById('rm-personal-name').textContent=`${profile.name}さんのロードマップ`;
  document.getElementById('rm-personal-sub').textContent=`${t.emoji} ${t.name} × ${c.label}`;
  document.getElementById('rm-personal-header').style.borderColor=t.color+'44';
  const offsets={basic:[0,6,12],mens:[0,6,12],eye:[0,3,8]};
  const assistantHTML=c.phases.map((p,i)=>{
    const off=(offsets[currentCourse]||[0,6,12])[i]||0;
    const cal=joinInfo?`<div style="font-size:10px;font-weight:700;color:white;background:${p.color};padding:2px 8px;border-radius:20px">${addMonths(joinInfo.year,joinInfo.month,off)} 〜</div>`:'';
    return`<div class="tl-phase"><div class="tl-left"><div class="tl-dot" style="background:${p.color}">${i+1}</div>${i<c.phases.length-1?`<div class="tl-line" style="background:${p.lc}"></div>`:''}</div><div class="tl-content"><div style="display:flex;align-items:center;gap:8px;flex-wrap:wrap;margin-bottom:3px"><div class="tl-period" style="color:${p.color};margin-bottom:0">${p.period}</div>${cal}</div><div class="tl-title">${p.title}</div><div class="tl-skills">${p.skills.map(s=>`<span class="tl-skill" style="background:${p.color}18;border:1px solid ${p.color}44;color:${p.color}">${s}</span>`).join('')}</div><ul class="tl-items">${p.items.map(it=>`<li>${it}</li>`).join('')}</ul></div></div>`;
  }).join('');
  const avgDebut=currentCourse==='eye'?12:18;
  const careerHTML=CAREER_STEPS[currentType].map((step,i)=>{
    const totalOff=avgDebut+step.months;
    const cal=joinInfo?(step.months===0?`<div style="font-size:10px;font-weight:700;color:white;background:${step.color};padding:2px 8px;border-radius:20px">${addMonths(joinInfo.year,joinInfo.month,avgDebut)}（平均）</div>`:`<div style="font-size:10px;font-weight:700;color:white;background:${step.color};padding:2px 8px;border-radius:20px">${addMonths(joinInfo.year,joinInfo.month,totalOff)}〜</div>`):'';
    const periodLabel=step.months===0?'デビュー':`デビュー+${step.months>=12?Math.floor(step.months/12)+'年'+(step.months%12?step.months%12+'ヶ月':''):step.months+'ヶ月'}`;
    return`<div class="tl-phase"><div class="tl-left"><div class="tl-dot" style="background:${step.color};font-size:16px">${step.icon}</div>${i<CAREER_STEPS[currentType].length-1?`<div class="tl-line" style="background:${step.color}44"></div>`:''}</div><div class="tl-content"><div style="display:flex;align-items:center;gap:8px;flex-wrap:wrap;margin-bottom:3px"><div class="tl-period" style="color:${step.color};margin-bottom:0">${periodLabel}</div>${cal}</div><div class="tl-title" style="color:${step.color}">${step.label}</div><div style="font-size:11px;line-height:1.7;color:var(--gray);font-weight:300;margin-top:4px">${step.desc}</div></div></div>`;
  }).join('');
  document.getElementById('timeline').innerHTML=(msg?`<div class="personal-msg-box">💬 ${msg}</div>`:'')+`<div style="font-size:10px;letter-spacing:.2em;color:var(--rose);font-weight:700;margin-bottom:10px">── アシスタント期 ──</div>`+assistantHTML+`<div style="font-size:10px;letter-spacing:.2em;color:${t.color};font-weight:700;margin:16px 0 10px">── デビュー後のキャリア ──</div>`+careerHTML+`<div class="tl-goal"><div class="tl-goal-lbl">🏆 ${profile.name}さんの目指す姿</div><div class="tl-goal-txt">${goal}</div></div>`;
}

function renderResult(){
  const score={S:0,T:0,C:0,L:0};
  [ans.q1,ans.q2,ans.q5].forEach(v=>{if(v&&score[v]!==undefined)score[v]++});
  let best='S',top=0;
  for(const[k,v] of Object.entries(score)){if(v>top){top=v;best=k}}
  currentType=best;
  const t=TYPES[best];
  document.getElementById('res-hero').style.background=t.heroGrad;
  document.getElementById('res-emoji').textContent=t.emoji;
  const nm=document.getElementById('res-name');nm.textContent=t.name;nm.style.color=t.color;
  const ct=document.getElementById('res-catch');ct.textContent=t.catch;ct.style.color=t.color;
  document.getElementById('res-sub').textContent=t.sub;
  document.getElementById('res-desc').textContent=t.desc;
  document.getElementById('res-tags').innerHTML=t.tags.map(g=>`<span class="tag">${g}</span>`).join('');
  document.getElementById('res-msg').textContent=t.msg;
  const hasMbti=Object.values(mbtiAns).every(v=>v!==null);
  let mbtiCode='';
  if(hasMbti){mbtiCode=mbtiAns.EI+mbtiAns.SN+mbtiAns.TF+mbtiAns.JP;document.getElementById('mbti-display').textContent=mbtiCode;document.getElementById('mbti-type-name').textContent=MBTI_NAMES[mbtiCode]?`— ${MBTI_NAMES[mbtiCode]}`:'';const msgs=MBTI_MSGS[best];document.getElementById('mbti-msg').textContent=(msgs.EI[mbtiAns.EI]||'')+'\n\n'+(msgs.TF[mbtiAns.TF]||'');document.getElementById('mbti-block').style.display='block';}
  currentCourse='basic';document.querySelectorAll('.course-card').forEach(c=>c.classList.remove('on'));document.getElementById('cc-basic').classList.add('on');
  renderTimeline();
  if(ans.q6==='real')document.getElementById('chip-real').style.fontWeight='700';
  else if(ans.q6==='online')document.getElementById('chip-online').style.fontWeight='700';
  // スタッフ用リンク生成（同じURLに?staff=1をつけるだけ）
  const staffUrl=`${STAFF_URL_BASE}?staff=1&type=${best}&name=${encodeURIComponent(profile.name)}&school=${encodeURIComponent(profile.school)}&grade=${encodeURIComponent(profile.grade)}&q3=${ans.q3||''}&q4=${ans.q4||''}&q6=${ans.q6||''}&mbti=${mbtiCode}`;
  document.getElementById('staff-link').href=staffUrl;
  document.getElementById('qr-card').style.display='block';
  showScr('result');
}

function resetAll(){
  Object.keys(ans).forEach(k=>ans[k]=null);Object.keys(mbtiAns).forEach(k=>mbtiAns[k]=null);
  profile.name='';profile.school='';profile.grade='';profile.joinMonth='';
  document.getElementById('inp-name').value='';document.getElementById('inp-school').value='';
  document.querySelectorAll('.grade-btn,.opt,.mbti-btn,.course-card').forEach(el=>el.classList.remove('on'));
  document.getElementById('cc-basic').classList.add('on');
  for(let i=1;i<=6;i++){const b=document.getElementById('btn-q'+i);if(b)b.disabled=true}
  document.getElementById('btn-mbti').disabled=true;document.getElementById('btn-profile').disabled=true;
  document.getElementById('mbti-block').style.display='none';
  document.getElementById('qr-card').style.display='none';
  document.getElementById('hdr').style.display='block';document.getElementById('prog').style.display='none';
  showScr('start');
}
